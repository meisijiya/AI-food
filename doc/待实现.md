RLock lock = redisson.getLock("feed:like:lock:" + postId + ":" + userId);
try {
    if (lock.tryLock(0, 3, TimeUnit.SECONDS)) {
        // 检查是否已点赞
        Boolean isLiked = redisTemplate.opsForSet().isMember("feed:like:" + postId, userId);
        if (!isLiked) {
            redisTemplate.opsForSet().add("feed:like:" + postId, String.valueOf(userId));
            redisTemplate.opsForValue().increment("feed:like:count:" + postId);
            // 更新 DB
            feedPost.setLikeCount(feedPost.getLikeCount() + 1);
        }
    }
} finally {
    lock.unlock();
}
